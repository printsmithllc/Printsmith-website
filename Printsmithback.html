(same as before) // File: /server/index.js const express = require('express'); const multer = require('multer'); const cors = require('cors'); const nodemailer = require('nodemailer'); const path = require('path'); require('dotenv').config(); const app = express(); app.use(cors()); app.use(express.json()); app.use('/uploads', express.static('uploads')); const storage = multer.diskStorage({ destination: 'uploads/', filename: (req, file, cb) => cb(null, Date.now() + path.extname(file.originalname)) }); const upload = multer({ storage }); app.post('/upload', upload.single('file'), (req, res) => { if (!req.file) return res.status(400).send('No file uploaded.'); res.json({ message: 'File uploaded', filename: req.file.filename }); }); app.post('/contact', async (req, res) => { const { name, email, message } = req.body; let transporter = nodemailer.createTransport({ service: 'gmail', auth: { user: process.env.EMAIL, pass: process.env.PASS } }); try { await transporter.sendMail({ from: email, to: process.env.EMAIL, subject: `3D Print Inquiry from ${name}`, text: message }); res.status(200).send('Message sent'); } catch (err) { console.error(err); res.status(500).send('Failed to send email'); } }); const PORT = process.env.PORT || 5000; app.listen(PORT, () => console.log(`Server running on port ${PORT}`)); // File: /.render.yaml databases: [] services: - type: web name: 3d-print-backend env: node buildCommand: cd server && npm install startCommand: node index.js envVars: - key: EMAIL value: your@email.com - key: PASS value: your-email-password autoDeploy: true - type: web name: 3d-print-frontend env: static buildCommand: cd client && npm install && npm run build staticPublishPath: client/build // File: /server/.env EMAIL=your@email.com PASS=your-email-password
